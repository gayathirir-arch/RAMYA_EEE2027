class Solution {
    public int minimumSum(int num) {

        int[] a = new int[4];

        a[0] = num % 10;
        num = num / 10;

        a[1] = num % 10;
        num = num / 10;

        a[2] = num % 10;
        num = num / 10;

        a[3] = num % 10;

        // sort the array
        java.util.Arrays.sort(a);

        int num1 = a[0] * 10 + a[2];
        int num2 = a[1] * 10 + a[3];

        return num1 + num2;
    }
}
